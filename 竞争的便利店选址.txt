#include<iostream>
using namespace std;

int main() {
	int n;
	cout << "请输入小区总数：" << endl;
	cin >> n;
	double b[1000];
	int c[1000] = { 0 };
	cout << "请输入每个小区咖啡店的收入" << endl;
	for (int i = 0; i < n; i++) {
		cin >> b[i];
	}
	for (int i = 0; i < n; i++) {
		if (i % 2==0) {
			cout << "Java的选择：";
			int k;
			cin >> k;
			if (c[k-1] == 0&&c[k-2]!=1&&c[k]!=1) {
				c[k-1] = 1;
			}
			else {
				cout << "选择不合理，请重新选择";
				i--;
			}
		}
		else {
			cout << "Qt的选择：";
			int k;
			cin >> k;
			if (c[k-1] == 0 && c[k - 2] != 2 && c[k] != 2) {
				c[k-1] = 2;
			}
			else {
				cout << "选择不合理，请重新选择";
				i--;
			}
		}
	}
	double J = 0, Q = 0;
	for (int i = 0; i < n; i++) {
		if (c[i] == 1) {
			J += b[i];
		}
		else if (c[i] == 2) {
			Q += b[i];
		}
	}
	printf("Java 可以赚取%f", J);
	cout << endl;
	printf("Qt 可以赚取%f", Q);
	if (J < Q) {
		cout <<endl<< "Qt赚得多";
	}
	else if (Q < J) {
		cout << endl << "Java赚得多";
	}
	else {
		cout << endl << "赚得一样多";
	}
}
